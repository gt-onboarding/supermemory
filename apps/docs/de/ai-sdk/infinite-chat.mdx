Infinite Chat bietet unbegrenzten Kontext für Chat-Anwendungen mit automatischer Speicherverwaltung.

<div id="setup">
  ## Einrichtung
</div>

```typescript
import { streamText } from "ai"

const infiniteChat = createAnthropic({
  baseUrl: 'https://api.supermemory.ai/v3/https://api.anthropic.com/v1',
  apiKey: 'your-provider-api-key',
  headers: {
    'x-supermemory-api-key': 'supermemory-api-key',
    'x-sm-conversation-id': 'conversation-id'
  }
})

const result = await streamText({
  model: infiniteChat("claude-3-sonnet"),
  messages: [
    { role: "user", content: "Hallo! Merke dir, dass ich TypeScript liebe." }
  ]
})
```

<div id="example-usage">
  ## Beispielanwendung
</div>

```typescript
import { streamText } from "ai"

const infiniteChat = createOpenAI({
  baseUrl: 'https://api.supermemory.ai/v3/https://api.openai.com/v1',
  apiKey: 'your-provider-api-key',
  headers: {
    'x-supermemory-api-key': 'supermemory-api-key',
    'x-sm-conversation-id': 'conversation-id'
  }
})

const result = await streamText({
  model: infiniteChat("gpt-5"),
  messages: [
    { role: "user", content: "Worüber haben wir gestern gesprochen?" }
  ]
})

return result.toAIStreamResponse()
```

interface ConfigWithProviderUrl {

}
