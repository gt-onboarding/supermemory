```typescript Typescript
import OpenAI from "openai";

/**
 * supermemory प्रॉक्सी के साथ OpenAI क्लाइंट इनिशियलाइज़ करें
 * @param {string} OPENAI_API_KEY - आपकी OpenAI API कुंजी
 * @param {string} SUPERMEMORY_API_KEY - आपकी supermemory API कुंजी
 * @returns {OpenAI} - कॉन्फ़िगर किया हुआ OpenAI क्लाइंट
 */
const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
  baseURL: "https://api.supermemory.ai/v3/https://api.openai.com/v1",
  headers: {
    "x-supermemory-api-key": process.env.SUPERMEMORY_API_KEY,
    "x-sm-user-id": "Your_users_id"
  },
});
```

```python Python
import openai
import os

# supermemory प्रॉक्सी के साथ OpenAI क्लाइंट कॉन्फ़िगर करें
openai.api_base = "https://api.supermemory.ai/v3/https://api.openai.com/v1"
openai.api_key = os.environ.get("OPENAI_API_KEY")  # आपकी सामान्य OpenAI कुंजी
openai.default_headers = {
    "x-supermemory-api-key": os.environ.get("SUPERMEMORY_API_KEY"),  # आपकी supermemory कुंजी
}

# असीमित संदर्भ के साथ एक चैट कम्प्लीशन बनाएँ
response = openai.ChatCompletion.create(
  model="gpt-5-nano",
  messages=[{"role": "user", "content": "Your message here"}]
)
```

<div id="error-handling">
  ## Error Handling
</div>

<Note>
  supermemory को विश्वसनीयता को सर्वोच्च प्राथमिकता देते हुए डिज़ाइन किया गया है। यदि supermemory प्रोसेसिंग पाइपलाइन में कोई समस्या आती है, तो सिस्टम स्वचालित रूप से आपके अनुरोध को सीधे LLM provider तक फॉरवर्ड करने पर वापस चला जाएगा, जिससे आपकी applications के लिए शून्य डाउनटाइम सुनिश्चित होगा।
</Note>

प्रत्येक response में diagnostic headers शामिल होते हैं जो प्रोसेसिंग के बारे में जानकारी देते हैं:

| Header                           | Description                                                             |
| -------------------------------- | ----------------------------------------------------------------------- |
| `x-supermemory-conversation-id`  | conversation thread के लिए unique identifier                            |
| `x-supermemory-context-modified` | यह दर्शाता है कि supermemory ने context को संशोधित किया है या नहीं ("true" या "false") |
| `x-supermemory-tokens-processed` | इस request में प्रोसेस किए गए tokens की संख्या                         |
| `x-supermemory-chunks-created`   | इस conversation से बनाए गए नए चंक की संख्या                            |
| `x-supermemory-chunks-deleted`   | हटाए गए चंक की संख्या (यदि कोई हो)                                     |
| `x-supermemory-docs-deleted`     | हटाए गए Documents की संख्या (यदि कोई हो)                                |

यदि कोई error होती है, तो एक अतिरिक्त header `x-supermemory-error` जोड़ा जाएगा, जिसमें यह विवरण होगा कि क्या गलत हुआ। भले ही supermemory में error आए, आपका अनुरोध फिर भी underlying LLM provider द्वारा प्रोसेस किया जाएगा।
