<div id="installation">
  ## Instalação
</div>

```sh
# instalar a partir do PyPI
pip install --pre supermemory
```


<div id="usage">
  ## Como usar
</div>

```python
import os
from supermemory import Supermemory

client = supermemory(
    api_key=os.environ.get("SUPERMEMORY_API_KEY"),  # Este é o padrão e pode ser omitido
)

response = client.search.documents(
    q="documents relacionados a Python",
)
print(response.results)
```


<div id="javascript-sdk">
  ## SDK JavaScript
</div>

<div id="installation">
  ## Instalação
</div>

```sh
npm install supermemory
```


<div id="usage">
  ## Como usar
</div>

```js
import supermemory from 'supermemory';

const client = new supermemory({
  apiKey: process.env['SUPERMEMORY_API_KEY'], // Este é o padrão e pode ser omitido
});

async function main() {
  const response = await client.search.documents({ q: 'documentos relacionados a Python' });

  console.debug(response.results);
}

main();
```
